\section{\textbf{等号}\textbf{(ColonEqual)}}
\subsection{\textbf{解説}}
\begin{maplelatex}\begin{Maple Normal}{
等号は，数学でいろいろな意味を持つことを中学校で学ぶ．等号それぞれの状況による操作の違いを人間は適当に判断できるが，プログラムであるMapleではそれぞれ違った記号や操作として用意され，人間がMapleに指示する必要がある．}\end{Maple Normal}
\end{maplelatex}
\subsubsection{\textbf{\textit{変数への代入:= (colonequal)}}}
\begin{maplegroup}
\begin{Maple Normal}{
変数に値を代入する時には:= (colonequal)を使う．例えば，}\end{Maple Normal}

\begin{center}
\begin{Maple Normal}{
a=3, b=2のとき，a+bはいくらか？}\end{Maple Normal}
\end{center}
\begin{Maple Normal}{
という問題を，Mapleで解かす時には，}\end{Maple Normal}

\begin{center}
\begin{Maple Normal}{
aに3, bに2を代入したとき，a+bはいくらか？}\end{Maple Normal}
\end{center}
\begin{Maple Normal}{
と読み直し，}\end{Maple Normal}

\textbf{a:=3; b:=2; a+b;}\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{a := 3}{\[\displaystyle a\, := \,3\]}
\end{maplelatex}
\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{b := 2}{\[\displaystyle b\, := \,2\]}
\end{maplelatex}
\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{5}{\[\displaystyle 5\]}
\end{maplelatex}
\end{maplegroup}
\begin{maplegroup}
\begin{Maple Normal}{
式の定義も同様．以下はa*x...=c*x\symbol{94}2...という式をeq1として定義している．}\end{Maple Normal}

\textbf{eq1:=a*x+b=c*x\symbol{94}2+d*x+e;}\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{eq1 := 3*x+2 = c*x^2+d*x+e}{\[\displaystyle {\it eq1}\, := \,3\,x+2=c{x}^{2}+dx+e\]}
\end{maplelatex}
\end{maplegroup}
\subsubsection{\textbf{\textit{変数の初期化(restart)}}}
\begin{maplegroup}
\begin{Maple Normal}{
一度何かを代入した変数を何も入れていない状態に戻す操作を変数の初期化という．すべての変数を一度に初期化するには，}\end{Maple Normal}

\textbf{restart;}\end{maplegroup}
\begin{maplegroup}
\begin{Maple Normal}{
とする．ひとつの変数だけを初期化するには，シングルクォート’でくくる．}\end{Maple Normal}

\textbf{a:='a';}\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{a := a}{\[\displaystyle a\, := \,a\]}
\end{maplelatex}
\end{maplegroup}
\subsubsection{\textbf{\textit{方程式の解(solve)}}}
\begin{maplegroup}
\begin{center}
\begin{Maple Normal}{
3x=2を満たすxをもとめよ．}\end{Maple Normal}
\end{center}
\begin{Maple Normal}{
という問題は，}\end{Maple Normal}

\textbf{solve(3*x=2,x);}\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{2/3}{\[\displaystyle 2/3\]}
\end{maplelatex}
\end{maplegroup}
\begin{maplegroup}
\begin{Maple Normal}{
連立方程式は以下のとおり．}\end{Maple Normal}

\textbf{solve(\{x+y=1,x-y=2\},\{x,y\});}\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{{x = 3/2, y = -1/2}}{\[\displaystyle  \left\{ x=3/2,y=-1/2 \right\} \]}
\end{maplelatex}
\end{maplegroup}
\begin{maplegroup}
\begin{Maple Normal}{
ただし，solveだけでは，x,yに値は代入されない．}\end{Maple Normal}

\textbf{sol1:=solve(\{x+y=1,x-y=2\},\{x,y\});}
\textbf{assign(sol1);}\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{sol1 := {x = 3/2, y = -1/2}}{\[\displaystyle {\it sol1}\, := \, \left\{ x=3/2,y=-1/2 \right\} \]}
\end{maplelatex}
\end{maplegroup}
\begin{maplegroup}
\begin{Maple Normal}{
とする必要がある．確認してみると}\end{Maple Normal}

\textbf{x,y;}\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{3/2, -1/2}{\[\displaystyle 3/2,\,-1/2\]}
\end{maplelatex}
\end{maplegroup}
\begin{maplegroup}
\begin{Maple Normal}{
となり，値が代入されていることがわかる．}\end{Maple Normal}

\end{maplegroup}

\begin{maplelatex}\begin{Maple Normal}{

\begin{maplelatex}\begin{Maple Normal}{

\end{maplelatex}
\begin{maplegroup}
\begin{Maple Normal}{

\end{center}
\begin{Maple Normal}{
と与えられるので，そのままMapleコマンドに読み替えて}\end{Maple Normal}

\textbf{expand((x-2)\symbol{94}2);}\mapleresult
\begin{maplelatex}
\mapleinline{inert}{2d}{x^2-4*x+4}{\[\displaystyle {x}^{2}-4\,x+4\]}
\end{maplelatex}
\end{maplegroup}
\begin{maplegroup}
\begin{Maple Normal}{
とすればよい．因数分解(factor)や微分(diff)・積分(int)も同様に等号で結ばれるが，Mapleには操作を指示する必要がある．詳しくは他の単元で．}\end{Maple Normal}

\end{maplegroup}
